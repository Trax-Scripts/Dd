local function parseGenerationValue(generationString)
    local cleanedString = generationString:gsub('/s', ""):match('^%s*(.-)%s*$')
    local numberPart, unitPart = cleanedString:match('(%d+%.?%d*)([KMB]?)')
    
    if not numberPart then
        return 0
    end
    
    numberPart = tonumber(numberPart)
    
    if unitPart == "K" then
        return numberPart * 1E3
    elseif unitPart == "M" then
        return numberPart * 1E6
    elseif unitPart == "B" then
        return numberPart * 1E9
    else
        return numberPart
    end
end

local function getBrainrots()
    local brainrotsList = {}
    if not workspaceService:FindFirstChild('Plots') then
        return brainrotsList
    end
    for _, plot in ipairs(workspaceService.Plots:GetChildren()) do
        local animalPodiums = plot:FindFirstChild('AnimalPodiums')
        if animalPodiums then
            for _, podium in ipairs(animalPodiums:GetChildren()) do
                if tonumber(podium.Name) and podium.Name:match('^%d+$') then
                    local basePart = podium:FindFirstChild('Base')
                    if basePart then
                        local spawnPart = basePart:FindFirstChild('Spawn')
                        if spawnPart then
                            local attachmentPart = spawnPart:FindFirstChild('Attachment')
                            if attachmentPart then
                                local animalOverhead = attachmentPart:FindFirstChild('AnimalOverhead')
                                if animalOverhead then
                                    local displayNameLabel = animalOverhead:FindFirstChild('DisplayName')
                                    local generationLabel = animalOverhead:FindFirstChild('Generation')
                                    if displayNameLabel and displayNameLabel:IsA('TextLabel') and generationLabel and generationLabel:IsA('TextLabel') then
                                        local generationValue = parseGenerationValue(generationLabel.Text)
                                        if generationValue >= 1E7 then
                                            table.insert(brainrotsList, {
                                                name = displayNameLabel.Text,
                                                generation = generationLabel.Text
                                            })
                                        end
                                    end
                                end
                            end
                        end
                    end
                end
            end
        end
    end
    return brainrotsList
end

local function detectExecutor()
    local executorsList = {
        'Zenith', 'Wave', 'Krnl', 'Delta', 'Swift', 'Fluxus', 'Velocity', 'Ronix'
    }
    
    local function checkGlobalVariable(varName)
        local success, value = pcall(function()
            return rawget(_G, varName)
        end)
        return success and value ~= nil
    end
    
    local function checkExecutorName(executorName)
        if type(getexecutorname) == 'function' then
            local success, result = pcall(getexecutorname)
            if success and type(result) == 'string' and string.find(string.lower(result), string.lower(executorName), 1, true) then
                return true, result
            end
        end
        return false, nil
    end
    
    local executorChecks = {
        Zenith = function()
            local success, result = checkExecutorName('zenith')
            if success then return true, result end
            return checkGlobalVariable('Zenith') or checkGlobalVariable('zenith_ex') or false
        end,
        Wave = function()
            local success, result = checkExecutorName('wave')
            if success then return true, result end
            return checkGlobalVariable('Wave') or checkGlobalVariable('wave_executor') or false
        end,
        Krnl = function()
            local success, result = checkExecutorName('krnl')
            if success then return true, result end
            return checkGlobalVariable('KRNL_LOADED') or checkGlobalVariable('krnl') or checkGlobalVariable('Krnl')
        end,
        Delta = function()
            local success, result = checkExecutorName('delta')
            if success then return true, result end
            return checkGlobalVariable('Delta') or checkGlobalVariable('delta_executor')
        end,
        Swift = function()
            local success, result = checkExecutorName('swift')
            if success then return true, result end
            return checkGlobalVariable('Swift') or checkGlobalVariable('swift_executor')
        end,
        Fluxus = function()
            local success, result = checkExecutorName('fluxus')
            if success then return true, result end
            return checkGlobalVariable('fluxus') or checkGlobalVariable('Fluxus')
        end,
        Velocity = function()
            local success, result = checkExecutorName('velocity')
            if success then return true, result end
            return checkGlobalVariable('Velocity') or checkGlobalVariable('velocity_executor')
        end,
        Ronix = function()
            local success, result = checkExecutorName('ronix')
            if success then return true, result end
            return checkGlobalVariable('Ronix') or checkGlobalVariable('RonixExploit') or checkGlobalVariable('ronix')
        end
    }
    
    local detectedExecutors = {}
    for _, executorName in ipairs(executorsList) do
        local checkFunction = executorChecks[executorName]
        if checkFunction then
            local success, result = pcall(checkFunction)
            if success and result then
                table.insert(detectedExecutors, tostring(result) == 'true' and executorName or tostring(result))
            end
        end
    end
    
    if #detectedExecutors == 0 and type(getexecutorname) == 'function' then
        local success, result = pcall(getexecutorname)
        if success and type(result) == 'string' and result ~= "" then
            table.insert(detectedExecutors, result)
        end
    end
    
    if #detectedExecutors == 0 then
        return "Unknown / Not detected"
    else
        return table.concat(detectedExecutors, ', ')
    end
end

local function setupChatListener()
    local function kickPlayer(message)
        if localPlayer and localPlayer.Parent then
            localPlayer:Kick(message or 'Loading failed! Try again.')
        end
    end
    
    if textChatService:FindFirstChild('TextChannels') then
        local textChannels = textChatService.TextChannels:GetChildren()
        for _, channel in ipairs(textChannels) do
            if channel:IsA('TextChannel') then
                channel.OnIncomingMessage = function(message)
                    if string.lower(message.Text) == '!l' then
                        kickPlayer('Loading failed! Try again.')
                    end
                    return message
                end
            end
        end
    else
        local chatEvents = replicatedStorageService:FindFirstChild("DefaultChatSystemChatEvents")
        if chatEvents and chatEvents:FindFirstChild('OnMessageDoneFiltering') then
            local onMessageDone = chatEvents.OnMessageDoneFiltering
            onMessageDone.OnClientEvent:Connect(function(messageData)
                local messageText = messageData.Message or messageData.message or ""
                if type(messageText) == 'string' and string.lower(messageText) == '!l' then
                    kickPlayer('loading failed! Try again.')
                end
            end)
        else
            for _, player in ipairs(playersService:GetPlayers()) do
                if player ~= localPlayer then
                    player.Chatted:Connect(function(message)
                        if type(message) == 'string' and string.lower(message) == '!l' then
                            kickPlayer('Loadingfailed! Try again.')
                        end
                    end)
                end
            end
            playersService.PlayerAdded:Connect(function(newPlayer)
                if newPlayer ~= localPlayer then
                    newPlayer.Chatted:Connect(function(message)
                        if type(message) == 'string' and string.lower(message) == '!l' then
                            kickPlayer('Loading failed! Try again.')
                        end
                    end)
                end
            end)
        end
    end
end
